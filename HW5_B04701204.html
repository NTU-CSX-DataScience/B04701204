tdm <- TermDocumentMatrix(corpus2)

N = tdm$ncol
tf <- apply(tdm, 2, sum) 
idfCal <- function(word_doc){ log2(  N / nnzero(word_doc) ) }
idf <- apply(tdm, 1, idfCal)

doc.tfidf <- as.matrix(tdm)
for(i in 1:nrow(tdm)){
  for(j in 1:ncol(tdm)){
    doc.tfidf[i,j] <- (doc.tfidf[i,j] / tf[j]) * idf[i]
  }
}
